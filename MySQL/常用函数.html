# 字符串函数

# concat

拼接字符串

select concat('小明同学','正在学习：','mysql课程');
1

# lower & upper & lcase & ucase

将字符串转为小写或者大写

select lower('HELLO WORLD');
1
select upper('hello world');
1
select lcase('HELLO WORLD');
1
select ucase('hello world');
1

# length & char_length

获取字符串长度

select length('hello');
1
select char_length('aaa');
1

# reverse

将字符串反转

select reverse('hello world');
1

# trim

将字符串两边的空格去除

select trim('      hello      ');
1

# ltrim & rtrim

将字符串左边或者右边的空格去除

select ltrim('    hello');
1
select rtrim('hello     ');
1

# replace

将字符串中的某个字符替换成其他字符

select replace('hello','h','X');
1

# repeat

将某个字符串或字符重复n次后返回

select repeat('a',10);
1

# lpad & rpad

将字符串的左边或右边使用第三个参数进行填充，直至整个字符串的长度为5

select lpad('aaa',5,'b');
1
select rpad('aaa',5,'b');
1

# right & left

将字符串的右边或者左边取指定数量的字符

select right('hello',2);
1
select left('hello',2);
1

# 常见使用案例

select concat(left(content,250),'...') from article;
1

# mid

将字符串从指定位置取字符，第二个参数为取字符起始位置，第三个参数为取字符的数量

select mid('this is demo',4,5);
1

# substring

将字符串从指定位置向右截取所有字符串

select substring('this is demo',3);
1

# space

生成指定数量的空格

select concat('left',space(20),'right');
1

# 数值函数

# ceil(ceiling) & floor

将返回大于n的最小整数和最大整数

select ceil(10.1); # 11
select ceiling(10.1); #11
1
2
select floor(10.1); # 10
1

# round

将保留几位小数，四舍五入

select round(3.3335,3); # 3.334
1

# truncate

将保留几位小数，不会四舍五入

select truncate(3.3335,3); # 3.333
1

# rand

将返回0~1的随机数

select rand();
1

# bin

将返回数值的二进制，十进制转二进制

select bin(10); # 1010
1

# max & min()

将结果集中取最大值和最小值

select max(id) from user;
1
select min(id) from user;
1

# sqrt

将返回数值的平方根(开平方)

select sqrt(100); # 10
1

# 日期与时间函数

# now

将返回当前的日期和时间

select now(); # 2020-07-23 15:59:18
1

# curdate & curtime

将返回当前日期或者当前的时间

select curdate(); # 2020-07-23
1
select curtime(); # 16:00:50
1

# year & month & day

将返回日期中的年、月、日

select year('2020-07-23'); # 2020
1
select month('2020-07-23'); # 7
1
select day('2020-07-23'); # 23
1

# hour & minute & second

将返回时间中的时、分、秒

select hour(now());
1
select minute(now());
1
select second(now());
1

# time

将返回日期中的时间部分

select time(now()); # 16:17:01
1

# quarter

将返回一年中的季度

select quarter(now());
1

# week

将返回一年中的第几周

select week(now());
1

# current_date

将返回当前日期

select current_date();
1

# current_time

将返回当前时间

select current_time();
1

# dayofyear

将返回一年中的第几天

select dayofyear(now());
1

# dayofmonth

将返回月份中的当前天数

select dayofmonth(now());
1

# dayofweek

将返回当前星期几，星期天(1)-星期六(7)

select dayofweek(now());
1

# weekday

将返回当前星期几，星期一(0)-星期天(6)

select weekday(now());
1

# unix_timestamp

将根据日期返回秒数

select unix_timestamp(now());
1

# last_day

将返回当前日期该月的最后一天

select last_day(now());
1

# addtime

将指定的时间添加时间，负数为减少，只能对时间设置

select addtime(now(),'-7:00:00'); # 返回七小时前的日期
1

# date_add

将指定的日期添加日期

select date_add(now(),interval 7 day); # 返回7天后的日期时间
1

支持时间单位有：year month day hour minute second

HOUR_MINUTE DAY_HOUR DAY_MINUTE DAY_SECOND HOUR_MINUTE HOUR_SECOND

select date_add(now(),interval '2 8' day_hour); # 返回2天8小时后的日期
1

# timestampdiff

将返回两个日期的间隔时间，间隔时间参数为 year month day hour minute second

select timestampdiff(day,'2020-01-01','2020-07-23'); # 204
1

# date_format

将返回格式化的日期

select date_format(now(),'%Y月%m月%d日-%H时%i分%s秒'); # 2020月07月23日-16时12分35秒
1

# from_unixtime

将时间戳转成日期

select from_unixtime(1595493756);
1

# datediff

将返回日期的差值

select datediff('2020-01-01','2020-07-23'); # -204
1
