# 连接操作

# 命令行连接MySQL服务器参数

参数 说明 默认
-u 数据库账号(username) 当前系统同名账号
-p 数据库账号密码(password)
-P 数据库端口(port) 3306
-h 数据库主机地址(host) 127.0.0.1/localhost
-e 执行SQL语句

在命令行输入账号密码登录并且执行SQL语句，但是不会进入MySQL控制台，登录执行完后就退出

mysql -uroot -proot -e"show databases;"
1

更改数据库连接端口和主机地址连接

mysql -uroot -P3308 -h 192.168.1.1
1

最常用的命令行连接MySQL控制台命令

mysql -uroot -p
1

退出MySQL控制台

exit;
1

每次执行一条SQL语句，我们会在后加上;结束这条SQL语句，写完发现不是我们要的SQL语句不想执行可以使用\c结束

show databases \c
1

# 数据库管理

创建一个shop数据库，字符集为utf8

create database shop charset utf8;
1

显示所有数据库名称

show databases;
1

切换到shop数据库

use shop;
1

删除shop数据库

drop databases shop;
1

如果shop数据库存在，则删除

drop databases if exists shop;
1

查看创建数据库的结构(语句)

show create databases shop;
1

# 数据库导入导出

# 导入数据库

将sql.sql的mysql语句登录执行

mysql -uroot -p < sql.sql
1

连接MySQL数据集后使用source命令导入执行sql.sql

mysql> source sql.sql;
1

# 导出数据库

使用mysqldump工具导出teacher数据库到指定的SQL文件

mysqldump -uroot -p teacher > teacher.sql
1

导出指定表的数据到SQL文件

mysqldump -uroot -p teacher userlist > teacher.sql
1

# 复制表数据

创建一个结构和userlist表结构一样的表(不会复制数据)

create table teacherlist like userlist;
1

复制userlist表数据到teacher表中

insert into teacher select * from userlist;
1

复制指定字段的数据到userlist表中

insert into teacherlist(username) select username from userlist;
1

复制指定字段，并且字段名不同则起别名进行复制数据

create table hd (id int primary key AUTO_INCREMENT,name varchar(30)) select id,cname as name from class;
1

复制表也复制数据

create table teacherlist select * from userlist;
1

# 临时表

临时表是用于储存临时数据表，会在数据库连接中断时自动删除。

  • 可以与普通表同名，优先级高于普通表
  • 连接中断时自动删除

创建一个临时表

create TEMPORARY table c_name select * from class_name;
1

删除临时表

drop temporary table if exists c_name;
1

# 查询数据

# 简单查询

查询所有字段的数据

select * from class;
1

查询指定字段的数据

select id,name,sex from class;
1

# 条件筛选查询

按照条件查询

select * from class where name = "alice";
1

按照包含指定字符的查询

select * from class where name like '%a%';
1

按照多条件查询

select * from class where id>1 and name = 'alice';
1

按照或条件查询

select * from class where id = 1 or name like '%汪%';
1

按照不包含指定字符的查询

select * from class where name not like '%汪%';
1

查询name字段第二个字符为u的学生

select * from students where name like '_u%';
1

like表达式中，%用于匹配多个字符，_用于匹配一个字符

按照去重复值去查询某个字段（与where条件配合查询无效）

select distinct name from class;
1

查询年龄在10~20岁间的学生

select * from class where age between 10 and 20;
1

查询年龄不在10~20岁间的学生

select * from class where age not between 10 and 20;
1

IN语句查询（查询10岁和20岁的所有学生）

select * from class where age in(10,20);
1

IN语句查询（查询除了10岁和20岁的所有学生）

select * from class where age not in(10,20);
1

# NULL

查询username字段值为null的数据

select * from userlist where username is null;
1

查询username字段值不为null的数据

select * from userlist where username is not null;
1

使用if语句判断是否为null，为null则显示未分配昵称，不是null则显示原本值

select if(username is null,'未分配昵称',username) from userlist;
1

上面的操作可使用ifnull函数简化

select ifnull(username,"未分配昵称") from userlist;
1

# 查询排序

以age列从大到小排序

select * from userlist order by age desc;
1

以age列从小到大排序

select * from userlist order by age asc;
1

获取随机一条数据

select * from userlist order by RAND() limit 1;
1

根据id值降序查询一条记录

select * from userlist order by id desc limit 1;
1

根据id从第二条到第四条升序查询

Limit 默认从0开始，第一个参数为从哪行开始查询默认为0，第二个参数为查询多少条数据

select * from userlist order by id asc limit 1,3;
1
