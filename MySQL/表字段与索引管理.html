# 修改表

# 修改表名

修改user表名为student

alter table user rename student;
1

另一种修改表名方法

rename table student to students;
1

# 修改表字符集

alter table student charset gbk;
1

# 清空表数据

truncate table students;
1

truncate清空表和delete from students;是有区别的

truncate执行后相当于重新建了一个表，所有的内容都是新表

delete执行后就像是手动一条一条的删除数据直到删完，自增不受影响

# 字段管理

# 修改表字段类型

alter table students modify name varchar(25) not null;
1

# 修改表字段以及字段名

alter table students change name sname varchar(30) not null;
1

# 添加表字段

alter table students add class varchar(100) not null;
1

# 在age字段后添加email字段

alter table students add email varchar(25) not null after age;
1

# 删除字段

alter table students drop email;
1

# 将字段添加到最前面第一个

alter table students add email varchar(25) not null first;
1

# 查看索引

show index from student;
show index from student\G; # 更详细的索引信息

# 主键索引

# 添加id字段

alter table student add id int not null first;
1

# 添加表主键

alter table student add PRIMARY KEY(id);
1
alter table student modify id int not null primary key;
1

# 删除表主键

alter table student drop PRIMARY KEY;
1

删除主键需要先删除自增，否则会报错

# 添加自增

alter table student modify id int not null AUTO_INCREMENT;
1

若id字段没有主键，执行这条SQL则会报错

ERROR 1075 (42000): Incorrect table definition; there can be only one auto column and it must be defined as a key
1

是因为MySQL要设置自增，则必须该字段为主键

主键索引标识符为 PRI

# 添加主键和自增

alter table student modify id int not null AUTO_INCREMENT, add PRIMARY KEY(id);
alter table student modify id int not null primary key auto_increment;
1
2

推荐这么使用

主键索引检索速度比唯一索引和普通索引都要快

# 重置自增

alter table student auto_increment = 1; # 前提是该表已经没有数据，再重置
truncate student; # 清空表数据，并且自动重置自增
1
2

# 唯一索引

唯一索引，不允许有重复值

唯一索引标识符为 UNI

# 创建和删除唯一索引

创建索引

create unique index u_cname on class(cname); # 为class表中的cname字段添加唯一索引并起别名为u_cname
1

删除索引

drop index u_cname on class;
1

# 添加唯一索引

alter table class add unique(cname); # 为cname字段添加唯一索引
alter table class add unique u_cname(cname); # 为cname字段添加唯一索引并起别名
1
2

# 删除唯一索引

alter table class drop index cname; # 删除cname字段的索引
alter table class drop index u_cname; # 删除别名为u_cname字段的索引
1
2

# 普通索引

索引用来提升查询速度，普通索引用的比较多

普通索引标识符为 MUL

# 创建和删除普通索引

创建索引

create index i_cname on class(cname); # 为class表中的cname字段添加普通索引并起别名为i_cname
1

删除索引

drop index i_cname on class;
1

# 添加普通索引

alter table class add index(cname); # 为cname字段添加普通索引
alter table class add index i_cname(cname); # 为cname字段添加普通索引并起别名
1
2

# 删除普通索引

alter table class drop index cname; # 删除cname字段的索引
alter table class drop index i_cname; # 删除别名为u_cname字段的索引
1
2

# 视图

视图是一个虚拟表，这个虚拟表是SQL查询的结果，当表中的数据发生变化时，视图数据也会随之发生变化。

# 创建视图

create view stuinfo as select stu.name,stu.age,class.name from stu,class where stu.class_id = class.id;
1

# 查看视图

select * from stuinfo;
1

# 删除视图

drop view stuinfo;
1
