# slice & substr & substring

截取字符串

let str = "hello world";
// 一个参数
console.log(str.slice(2)); // llo world 从第二个位置开始截取到最后
console.log(str.substr(2)); // llo world 从第二个位置开始截取到最后
console.log(str.substring(2)); // llo world 从第二个位置开始截取到最后
// 两个参数
console.log(str.slice(2, 7)); // llo w 第一个参数为起始位置(包括自己)截取到下标为第二个参数的位置
console.log(str.substr(2, 3)); // llo 第一个参数为起始位置(包括自己)截取到下标为第二个参数的位置
console.log(str.substring(2, 8)); // llo wo 第一个参数为起始位置(包括自己)截取到下标为第二个参数(不包括自己)的位置
// 负数
console.log(str.slice(4, -2)); // o wor 从右往左第一个参数(包括自己)起始截取4个字符
console.log(str.slice(-2)); // ld 从右往左-2起始位置往右截取所有
console.log(str.substr(-3, 2)); // rl 从右往左第一个参数(不包括自己)起始截取2个字符
console.log(str.substring(2, -4)); // he 此函数负数会转为0

# 自定义格式化日期函数

function DateFormat(date, date_format = "YYYY-MM-DD HH:mm:ss") {
    let config = {
        YYYY: date.getFullYear(),
        MM: date.getMonth() + 1,
        DD: date.getDate(),
        HH: date.getHours(),
        mm: date.getMinutes(),
        ss: date.getSeconds()
    };
    for (let key in config) {
        date_format = date_format.replace(key, config[key]);
    }
    return date_format;
}
console.log(DateFormat(new Date(), 'YYYY年MM月DD日')); // 2020年7月30日
